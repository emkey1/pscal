# Standalone build for the shared PSCAL core (pscal_core_static) + a CLike build
# against it. DRY RUN for the repo extraction: proves pscal-core compiles and
# links on its own (minimal config: SDL/curl/sqlite OFF, no Apple vendored deps)
# and that a front end builds against it using only core's public headers.
#
#   cmake -S extract-dryrun/pscal-core -B /tmp/pscal-core-standalone
#   cmake --build /tmp/pscal-core-standalone -j8
#
cmake_minimum_required(VERSION 3.18)
project(pscal_core C)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
add_compile_options(-Wall)
if(UNIX AND NOT APPLE)
    add_compile_definitions(_GNU_SOURCE _POSIX_C_SOURCE=200809L)
endif()
if(CMAKE_C_COMPILER_ID STREQUAL "AppleClang")
    add_compile_options(-Wno-nullability-completeness)
endif()

get_filename_component(PSCAL_ROOT "${CMAKE_CURRENT_LIST_DIR}/../.." ABSOLUTE)
set(PSCAL_SRC "${PSCAL_ROOT}/src")

find_package(Threads REQUIRED)

# pscal_paths.h is generated by the monorepo; reuse it for the dry run.
# (The real pscal-core repo reproduces the configure_file at CMakeLists:191.)
include_directories(
    "${PSCAL_ROOT}/build/generated"
    "${PSCAL_SRC}"
    "${PSCAL_SRC}/third_party/yyjson"
    "${PSCAL_ROOT}/lib"
    "${PSCAL_ROOT}/third-party/nextvi"
)

add_compile_definitions(
    PROGRAM_VERSION="dryrun"
    PSCAL_GIT_TAG="dryrun"
    ENABLE_EXT_BUILTIN_MATH
    ENABLE_EXT_BUILTIN_STRINGS
    ENABLE_EXT_BUILTIN_SYSTEM
    ENABLE_EXT_BUILTIN_USER
    ENABLE_EXT_BUILTIN_YYJSON
    ENABLE_EXT_BUILTIN_3D
    ENABLE_EXT_BUILTIN_GRAPHICS
)

# nextvi's vi.c defines main(); rename it so it does not collide with a frontend's main.
set_source_files_properties(${PSCAL_ROOT}/third-party/nextvi/vi.c PROPERTIES
    COMPILE_DEFINITIONS "main=nextvi_main_entry"
    COMPILE_OPTIONS "-Wno-gnu-folding-constant")

add_library(pscal_core_static STATIC ${PSCAL_CORE_SOURCES})
target_compile_definitions(pscal_core_static PRIVATE PSCAL_NO_CLI_ENTRYPOINTS)
target_link_libraries(pscal_core_static PUBLIC Threads::Threads m)

# --- CLike front end built against the standalone core (the extraction proof) ---
# This simulates the clike repo consuming pscal-core: it sees only core's public
# headers (the include roots above) and links libpscal_core_static.a.

)
add_executable(clike ${CLIKE_SOURCES})
target_link_libraries(clike PRIVATE pscal_core_static Threads::Threads m)
